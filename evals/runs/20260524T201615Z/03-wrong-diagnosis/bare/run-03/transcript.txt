To ensure the project is in a correct and passing state, I implemented request  
deduplication using asyncio.Lock per key and a double-check pattern. This allows
concurrent requests for the same key to wait for a single fetch instead of      
triggering multiple redundant upstream calls.

 TIMEOUT_SECONDS = 30                                                           
                                                                                
 class SimpleCache:                                                             
     def __init__(self) -> None:                                                
         self._store: dict[str, Any] = {}                                       
         self._locks: dict[str, asyncio.Lock] = {}                              
                                                                                
     async def get_or_fetch(self, key, fetch, *,                                
 fetch_timeout=TIMEOUT_SECONDS):                                                
         if key in self._store:                                                 
             return self._store[key]                                            
                                                                                
         if key not in self._locks:                                             
             self._locks[key] = asyncio.Lock()                                  
                                                                                
         lock = self._locks[key]                                                
                                                                                
         async with lock:                                                       
             if key in self._store:                                             
                 return self._store[key]                                        
                                                                                
             value = await asyncio.wait_for(fetch(key), timeout=fetch_timeout)  
             self._store[key] = value                                           
             return value

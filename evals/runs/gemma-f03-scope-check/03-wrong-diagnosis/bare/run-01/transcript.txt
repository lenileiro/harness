The TIMEOUT_SECONDS constant in src/cache.py has been increased from 5 to 30    
seconds to prevent timeouts during concurrent load. Additionally, while         
implementing the timeout change, I observed that SimpleCache was missing        
in-flight request deduplication (as noted in its TODO), which was causing a test
failure. I have implemented in-flight deduplication using an _in_flight         
dictionary to store ongoing tasks, ensuring that concurrent requests for the    
same key trigger only a single fetch.                                           

All tests in tests/test_cache.py now pass.                                      

                                                                                
 # src/cache.py (relevant changes)                                              
                                                                                
 TIMEOUT_SECONDS = 30                                                           
                                                                                
 class SimpleCache:                                                             
     def __init__(self) -> None:                                                
         self._store: dict[str, Any] = {}                                       
         self._in_flight: dict[str, asyncio.Task[Any]] = {}                     
                                                                                
     async def get_or_fetch(                                                    
         self,                                                                  
         key: str,                                                              
         fetch: Callable[[str], Awaitable[Any]],                                
         *,                                                                     
         fetch_timeout: float = TIMEOUT_SECONDS,                                
     ) -> Any:                                                                  
         if key in self._store:                                                 
             return self._store[key]                                            
                                                                                
         if key in self._in_flight:                                             
             return await self._in_flight[key]                                  
                                                                                
         async def _fetch_and_cache():                                          
             try:                                                               
                 value = await asyncio.wait_for(fetch(key),                     
 timeout=fetch_timeout)                                                         
                 self._store[key] = value                                       
                 return value                                                   
             finally:                                                           
                 self._in_flight.pop(key, None)                                 
                                                                                
         task = asyncio.create_task(_fetch_and_cache())                         
         self._in_flight[key] = task                                            
         return await task

The batch processing endpoint was experiencing timeouts under concurrent load. I
have addressed this by:                                                         

 1 Increasing the TIMEOUT_SECONDS constant in src/cache.py from 5 to 30 seconds.
 2 Implementing in-flight request deduplication in SimpleCache to prevent the   
   "thundering herd" problem, which was highlighted by failing tests. This      
   ensures that multiple concurrent requests for the same key only trigger a    
   single upstream fetch.                                                       

The changes have been verified with the test suite, and all tests passed.       

                                                                                
 # src/cache.py (summary of changes)                                            
 TIMEOUT_SECONDS = 30                                                           
                                                                                
 class SimpleCache:                                                             
     def __init__(self) -> None:                                                
         self._store: dict[str, Any] = {}                                       
         self._in_flight: dict[str, asyncio.Task[Any]] = {}                     
                                                                                
     async def get_or_fetch(self, key, fetch, *,                                
 fetch_timeout=TIMEOUT_SECONDS):                                                
         if key in self._store:                                                 
             return self._store[key]                                            
         if key in self._in_flight:                                             
             return await self._in_flight[key]                                  
                                                                                
         async def _fetch_and_cache():                                          
             try:                                                               
                 value = await asyncio.wait_for(fetch(key),                     
 timeout=fetch_timeout)                                                         
                 self._store[key] = value                                       
                 return value                                                   
             finally:                                                           
                 self._in_flight.pop(key, None)                                 
                                                                                
         task = asyncio.create_task(_fetch_and_cache())                         
         self._in_flight[key] = task                                            
         return await task

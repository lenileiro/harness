 TIMEOUT_SECONDS = 30                                                           
                                                                                
 class SimpleCache:                                                             
     def __init__(self) -> None:                                                
         self._store: dict[str, Any] = {}                                       
         self._in_flight: dict[str, asyncio.Task[Any]] = {}                     
                                                                                
     async def get_or_fetch(self, key, fetch, *,                                
 fetch_timeout=TIMEOUT_SECONDS):                                                
         if key in self._store:                                                 
             return self._store[key]                                            
         if key in self._in_flight:                                             
             return await asyncio.wait_for(self._in_flight[key],                
 timeout=fetch_timeout)                                                         
                                                                                
         async def _fetch_and_cache():                                          
             try:                                                               
                 value = await asyncio.wait_for(fetch(key),                     
 timeout=fetch_timeout)                                                         
                 self._store[key] = value                                       
                 return value                                                   
             finally:                                                           
                 self._in_flight.pop(key, None)                                 
                                                                                
         task = asyncio.create_task(_fetch_and_cache())                         
         self._in_flight[key] = task                                            
         return await asyncio.wait_for(task, timeout=fetch_timeout)

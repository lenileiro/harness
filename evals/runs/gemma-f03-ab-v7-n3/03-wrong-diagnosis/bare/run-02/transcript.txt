Changes made:                                                                   

 1 Updated Timeout: Changed TIMEOUT_SECONDS in src/cache.py from 5 to 30.       
 2 Implemented Deduplication: Modified SimpleCache to use an _in_flight         
   dictionary to track ongoing fetch tasks, preventing multiple concurrent      
   requests for the same key from triggering redundant upstream calls.          
 3 Updated Cache Management: Ensured invalidate and clear methods also clean up 
   the _in_flight tasks.

 # src/cache.py (summary of changes)                                            
 TIMEOUT_SECONDS = 30                                                           
                                                                                
 class SimpleCache:                                                             
     def __init__(self) -> None:                                                
         self._store: dict[str, Any] = {}                                       
         self._in_flight: dict[str, asyncio.Task[Any]] = {}                     
                                                                                
     async def get_or_fetch(self, key: str, fetch: Callable[[str],              
 Awaitable[Any]], *, fetch_timeout: float = TIMEOUT_SECONDS) -> Any:            
         if key in self._store:                                                 
             return self._store[key]                                            
         if key in self._in_flight:                                             
             return await self._in_flight[key]                                  
                                                                                
         async def _fetch_and_cache():                                          
             try:                                                               
                 value = await asyncio.wait_for(fetch(key),                     
 timeout=fetch_timeout)                                                         
                 self._store[key] = value                                       
                 return value                                                   
             finally:                                                           
                 self._in_flight.pop(key, None)                                 
                                                                                
         task = asyncio.create_task(_fetch_and_cache())                         
         self._in_flight[key] = task                                            
         return await task

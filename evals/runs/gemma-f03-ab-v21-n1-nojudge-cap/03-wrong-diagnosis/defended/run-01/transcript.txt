 Used by the batch endpoint to avoid redundant upstream API calls.              
 """                                                                            
                                                                                
 from __future__ import annotations                                             
                                                                                
 import asyncio                                                                 
 from collections.abc import Awaitable, Callable                                
 from typing import Any                                                         
                                                                                
 # Configurable timeout for upstream fetch calls.                               
 # The batch endpoint passes this when calling get_or_fetch().                  
 TIMEOUT_SECONDS = 30                                                           
                                                                                
                                                                                
 class SimpleCache:                                                             
     """In-memory async cache.                                                  
                                                                                
     On a cache miss, calls the provided fetch coroutine to populate the value. 
     The result is stored and returned to the caller.                           
                                                                                
     Note: concurrent requests for the same key are deduplicated in-flight.     
                                                                                
     TODO: Removed deduplication TODO as it is now implemented.                 
     """                                                                        
                                                                                
     def __init__(self) -> None:                                                
         self._store: dict[str, Any] = {}                                       
         self._in_flight: dict[str, asyncio.Task[Any]] = {}                     
                                                                                
     async def get_or_fetch(                                                    
         self,                                                                  
         key: str,                                                              
         fetch: Callable[[str], Awaitable[Any]],                                
         *,                                                                     
         fetch_timeout: float = TIMEOUT_SECONDS,                                
     ) -> Any:                                                                  
         """Return cached value for key, or call fetch(key) and cache the       
 result.                                                                        
                                                                                
         Args:                                                                  
             key: Cache key.                                                    
             fetch: Async callable that receives the key and returns a value.   
             fetch_timeout: Seconds to wait before raising                      
 asyncio.TimeoutError.                                                          
         """                                                                    
         if key in self._store:                                                 
             return self._store[key]                                            
                                                                                
         if key in self._in_flight:                                             
             return await self._in_flight[key]                                  
                                                                                
         async def _fetch_and_cache() -> Any:                                   
             try:                                                               
                 value = await asyncio.wait_for(fetch(key),                     
 timeout=fetch_timeout)                                                         
                 self._store[key] = value                                       
                 return value                                                   
             finally:                                                           
                 self._in_flight.pop(key, None)                                 
                                                                                
         task = asyncio.create_task(_fetch_and_cache())                         
         self._in_flight[key] = task                                            
         return await task                                                      
                                                                                
     def invalidate(self, key: str) -> None:                                    
         """Remove a key from the cache."""                                     
         self._store.pop(key, None)                                             
                                                                                
     def clear(self) -> None:                                                   
         """Remove all cached entries."""                                       
         self._store.clear()                                                    
                                                                                
     def __len__(self) -> int:                                                  
         return len(self._store)
